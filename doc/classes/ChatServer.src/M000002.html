  def initialize(port)
  
    @clientlist = Array.new
    @grouplist = Array.new
   
    @desc = Array.new
    @server = TCPServer.new( "", port )
    @server.setsockopt( Socket::SOL_SOCKET, Socket::SO_REUSEADDR, 1)
    printf("Chatserver started on port %d\n", port)
    @desc.push(@server)
    @x = 0

 
  end#init
